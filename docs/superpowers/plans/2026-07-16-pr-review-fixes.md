```go
// getRealPods queries the typed client for pods matching the given label selector.
// Returns PodInfo for each pod found. If the query fails (config load, clientset
// creation, or pod list), the error is returned so the caller can log it.
func getRealPods(ctx context.Context, namespace, selector string) ([]client.PodInfo, error) {
	config, err := client.LoadRestConfig(kubeconfig, kubeContext)
	if err != nil {
		return nil, fmt.Errorf("failed to load rest config: %w", err)
	}
	clientset, err := kubernetes.NewForConfig(config)
	if err != nil {
		return nil, fmt.Errorf("failed to create clientset: %w", err)
	}

	pods, err := clientset.CoreV1().Pods(namespace).List(
		ctx,
		metav1.ListOptions{LabelSelector: selector},
	)
	if err != nil {
		return nil, fmt.Errorf("failed to list pods: %w", err)
	}

	var result []client.PodInfo
	for _, pod := range pods.Items {
		result = append(result, client.PodInfo{
			Name:   pod.Name,
			Status: podDisplayStatus(&pod),
			IP:     pod.Status.PodIP,
			Node:   pod.Spec.NodeName,
		})
	}
	return result, nil
}
```

In `pkg/cli/lifecycle.go`, update the `createJobResources` function. Add a deferred rollback that deletes the CRD if the function returns an error. The function already returns errors at multiple points; the defer catches all of them.

Replace the function body (lines 20-66) with:

```go
func createJobResources(ctx context.Context, crd *unstructured.Unstructured, t *task.Task, k8sClient *client.Client, p provider.Provider) error {
	// Re-fetch CRD to get UID assigned by the API server
	created, err := k8sClient.Get(ctx, crd.GetKind(), crd.GetNamespace(), crd.GetName())
	if err != nil {
		return fmt.Errorf("failed to re-fetch CRD for UID: %w", err)
	}

	ownerRef := metav1.OwnerReference{
		APIVersion:         crd.GetAPIVersion(),
		Kind:               crd.GetKind(),
		Name:               crd.GetName(),
		UID:                created.GetUID(),
		BlockOwnerDeletion: ptrBool(true),
		Controller:         ptrBool(true),
	}

	// Rollback: if any auxiliary resource creation fails, delete the CRD
	// to prevent a partially-submitted job from being left in the cluster.
	success := false
	defer func() {
		if !success {
			if delErr := k8sClient.Delete(ctx, crd.GetKind(), crd.GetNamespace(), crd.GetName()); delErr != nil {
				log.Error(delErr, "failed to delete CRD during rollback",
					"kind", crd.GetKind(), "name", crd.GetName())
			}
		}
	}()

	// Create ConfigMap anchor storing the arena-v2.yaml task configuration
	yamlContent, err := yaml.Marshal(t)
	if err != nil {
		return fmt.Errorf("failed to marshal task configuration: %w", err)
	}
	cm := buildConfigMap(crd.GetName(), crd.GetNamespace(), string(yamlContent), ownerRef)
	if err := k8sClient.Create(ctx, cm); err != nil {
		return fmt.Errorf("failed to create ConfigMap: %w", err)
	}

	// Create RBAC resources if the provider produces any
	rbacResources, err := p.BuildRBAC(t, ownerRef)
	if err != nil {
		return fmt.Errorf("failed to build RBAC resources: %w", err)
	}
	for _, r := range rbacResources {
		r.SetNamespace(crd.GetNamespace())
		if err := k8sClient.Create(ctx, r); err != nil {
			return err
		}
	}

	// Create TensorBoard resources if enabled
	if t.Logging.TensorBoard != nil && t.Logging.TensorBoard.Enabled {
		if err := createTensorBoardResources(ctx, crd, t, k8sClient, ownerRef); err != nil {
			return err
		}
	}

	success = true
	return nil
}
```

Key changes:
- Added `success := false` flag and `defer` cleanup block after `ownerRef` is built
- The defer deletes the CRD if `success` is still false when the function returns
- Set `success = true` just before the final `return nil`
- Added `"github.com/kubeflow/arena/pkg/log"` to imports (for rollback error logging)

Also update the imports in `lifecycle.go` to add:
```go

// cleanupCRD deletes a CRD by kind/name/namespace using kubectl.
```
